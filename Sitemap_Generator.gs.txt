// ============================================================
// doGet — Serve sitemap as XML (deploy as Web App)
// ============================================================
function doGet(e) {
  const type = e?.parameter?.type || 'news';

  if (type === 'news') {
    const xml = generateNewsSitemap();
    return ContentService.createTextOutput(xml)
      .setMimeType(ContentService.MimeType.XML);
  }

  if (type === 'main') {
    const xml = generateMainSitemap();
    return ContentService.createTextOutput(xml)
      .setMimeType(ContentService.MimeType.XML);
  }

  return ContentService.createTextOutput('Invalid type')
    .setMimeType(ContentService.MimeType.TEXT);
}

// ============================================================
// GENERATE NEWS SITEMAP (Google News format)
// ============================================================
function generateNewsSitemap() {
  const articles = fetchArticles(500);
  const now      = new Date().toISOString();

  let xml = `<?xml version="1.0" encoding="UTF-8"?>
<urlset
  xmlns="http://www.sitemaps.org/schemas/sitemap/0.9"
  xmlns:news="http://www.google.com/schemas/sitemap-news/0.9"
  xmlns:image="http://www.google.com/schemas/sitemap-image/1.1">
`;

  articles.forEach(function(a) {
    if (!a.title || !a.id) return;

    // Build clean URL
    const url = SITE_BASE + '/newstallyopen.html'
      + '?nid=' + encodeURIComponent(a.id)
      + '&q='   + encodeURIComponent((a.title || '').substring(0, 100))
      + '&cat='  + encodeURIComponent(a.category || '')
      + '&src='  + encodeURIComponent(a.source || '');

    // Format date for Google News (must be within 2 days for News sitemap)
    var pubDate = '';
    if (a.date) {
      try {
        pubDate = new Date(a.date).toISOString();
      } catch(e) {
        pubDate = now;
      }
    } else {
      pubDate = now;
    }

    // Escape XML special chars
    const title  = xmlEscape(a.title || '');
    const source = xmlEscape(a.source || 'NewsTally');
    const cat    = xmlEscape(a.category || 'General');
    const lang   = a.language || 'en';
    const image  = a.image || '';

    xml += `  <url>
    <loc>${xmlEscape(url)}</loc>
    <lastmod>${pubDate}</lastmod>
    <changefreq>never</changefreq>
    <priority>0.8</priority>
    <news:news>
      <news:publication>
        <news:name>NewsTally</news:name>
        <news:language>${lang}</news:language>
      </news:publication>
      <news:publication_date>${pubDate}</news:publication_date>
      <news:title>${title}</news:title>
      <news:keywords>${cat}</news:keywords>
    </news:news>
${image ? `    <image:image>
      <image:loc>${xmlEscape(image)}</image:loc>
      <image:title>${title}</image:title>
      <image:caption>${title} — ${source}</image:caption>
    </image:image>` : ''}
  </url>
`;
  });

  xml += '</urlset>';
  return xml;
}

// ============================================================
// GENERATE MAIN SITEMAP
// ============================================================
function generateMainSitemap() {
  const now = new Date().toISOString();
  return `<?xml version="1.0" encoding="UTF-8"?>
<sitemapindex xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">
  <sitemap>
    <loc>${SITE_BASE}/sitemap.xml</loc>
    <lastmod>${now}</lastmod>
  </sitemap>
  <sitemap>
    <loc>${SITE_BASE}/sitemap-news.xml</loc>
    <lastmod>${now}</lastmod>
  </sitemap>
</sitemapindex>`;
}

// ============================================================
// FETCH ARTICLES FROM FIRESTORE
// ============================================================
function fetchArticles(maxCount) {
  var token  = getToken();
  var articles = [];
  var pageToken = null;
  var fetched   = 0;

  do {
    var url = 'https://firestore.googleapis.com/v1/projects/' + FIREBASE_PROJECT
      + '/databases/(default)/documents/news'
      + '?pageSize=100'
      + '&orderBy=savedAt+desc'
      + (pageToken ? '&pageToken=' + pageToken : '');

    var res = UrlFetchApp.fetch(url, {
      headers: { Authorization: 'Bearer ' + token },
      muteHttpExceptions: true
    });

    if (res.getResponseCode() !== 200) break;

    var data = JSON.parse(res.getContentText());
    var docs = data.documents || [];

    docs.forEach(function(d) {
      var f = d.fields || {};
      articles.push({
        id:       getStr(f.id)       || d.name.split('/').pop(),
        title:    getStr(f.title)    || getStr(f.headline) || '',
        source:   getStr(f.source)   || '',
        date:     getStr(f.date)     || getStr(f.savedAt)  || '',
        category: getStr(f.category) || 'General',
        image:    getStr(f.image)    || '',
        language: getStr(f.language) || 'en',
        url:      getStr(f.url)      || ''
      });
    });

    fetched   += docs.length;
    pageToken  = data.nextPageToken || null;

  } while (pageToken && fetched < maxCount);

  return articles;
}

function getStr(field) {
  if (!field) return '';
  return field.stringValue || field.integerValue || '';
}

// ============================================================
// SAVE SITEMAP TO GITHUB (via GitHub API)
// — Run manually or on trigger
// ============================================================
function saveSitemapToGitHub() {
  // Fill these:
  var GITHUB_TOKEN = 'YOUR_GITHUB_TOKEN'; // Settings → Developer Settings → PAT
  var GITHUB_REPO  = 'YOUR_USERNAME/YOUR_REPO'; // e.g. shivank/newstally
  var FILE_PATH    = 'sitemap-news.xml';

  var xml     = generateNewsSitemap();
  var encoded = Utilities.base64Encode(xml, Utilities.Charset.UTF_8);

  // Get current file SHA (needed for update)
  var shaRes = UrlFetchApp.fetch(
    'https://api.github.com/repos/' + GITHUB_REPO + '/contents/' + FILE_PATH,
    { headers: { Authorization: 'token ' + GITHUB_TOKEN }, muteHttpExceptions: true }
  );
  var sha = '';
  if (shaRes.getResponseCode() === 200) {
    sha = JSON.parse(shaRes.getContentText()).sha || '';
  }

  // Push file
  var payload = {
    message: 'Auto-update sitemap-news.xml ' + new Date().toISOString(),
    content: encoded
  };
  if (sha) payload.sha = sha;

  var res = UrlFetchApp.fetch(
    'https://api.github.com/repos/' + GITHUB_REPO + '/contents/' + FILE_PATH,
    {
      method:      'PUT',
      contentType: 'application/json',
      headers:     { Authorization: 'token ' + GITHUB_TOKEN },
      payload:     JSON.stringify(payload),
      muteHttpExceptions: true
    }
  );

  Logger.log('GitHub push: ' + res.getResponseCode());
  if (res.getResponseCode() === 200 || res.getResponseCode() === 201) {
    Logger.log('sitemap-news.xml updated on GitHub!');
  } else {
    Logger.log('Error: ' + res.getContentText().substring(0, 300));
  }
}

// ============================================================
// AUTO TRIGGER — Run daily
// ============================================================
function setupSitemapTrigger() {
  ScriptApp.getProjectTriggers().forEach(function(t) {
    if (t.getHandlerFunction() === 'saveSitemapToGitHub') {
      ScriptApp.deleteTrigger(t);
    }
  });
  ScriptApp.newTrigger('saveSitemapToGitHub')
    .timeBased().everyHours(6).create();
  Logger.log('Sitemap trigger set: every 6 hours');
}

// ============================================================
// FIREBASE TOKEN
// ============================================================
function getToken() {
  var cached = CacheService.getScriptCache().get('fb_sitemap_token');
  if (cached) return cached;

  var now   = Math.floor(Date.now() / 1000);
  var hdr   = Utilities.base64EncodeWebSafe(JSON.stringify({alg:'RS256',typ:'JWT'}));
  var clm   = Utilities.base64EncodeWebSafe(JSON.stringify({
    iss: SA.client_email,
    scope: 'https://www.googleapis.com/auth/datastore',
    aud: 'https://oauth2.googleapis.com/token',
    iat: now, exp: now + 3600
  }));
  var toSign = hdr + '.' + clm;
  var key    = SA.private_key.replace(/\\n/g, '\n');
  var sig    = Utilities.base64EncodeWebSafe(Utilities.computeRsaSha256Signature(toSign, key));
  var res    = UrlFetchApp.fetch('https://oauth2.googleapis.com/token', {
    method: 'post', contentType: 'application/x-www-form-urlencoded',
    payload: 'grant_type=urn:ietf:params:oauth:grant-type:jwt-bearer&assertion=' + toSign + '.' + sig
  });
  var token = JSON.parse(res.getContentText()).access_token;
  CacheService.getScriptCache().put('fb_sitemap_token', token, 3300);
  return token;
}

function xmlEscape(s) {
  return String(s || '')
    .replace(/&/g,  '&amp;')
    .replace(/</g,  '&lt;')
    .replace(/>/g,  '&gt;')
    .replace(/"/g,  '&quot;')
    .replace(/'/g,  '&apos;');
}
